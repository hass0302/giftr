function verifyAddPersonBtn() {
    let inputName = document.getElementById("inputName");
    let inputDate = document.getElementById("inputDate");
    let errorFlagName, errorFlagDate = true;
    // Verify if the fileds are valid. for names only to check if its blank
    if (inputName.value == "" || inputName.value == null) {
        let inputLabel = document.querySelector("#inputNamelbl");
        inputLabel.style = "color: red; font-weight: bold;";
        errorFlagName = true;
    }
    else {
        let inputLabel = document.querySelector("#inputNamelbl");
        inputLabel.style = "";
        errorFlagName = false;
    }
    // verify if the dates are good. 
    if (inputDate.value == "" || inputDate.value == null) {
        let inputLabel = document.getElementById("inputDOBlbl");
        inputLabel.style = "color: red; font-weight: bold;";
        errorFlagDate = true;
    }
    else {
        let inputLabel = document.querySelector("#inputDOBlbl");
        inputLabel.style = "";
        errorFlagDate = false;
    }
    if (!errorFlagDate && !errorFlagName) {
        data.people.push({
            "id": data.people.length
            , "name": inputName.value
            , "dob": inputDate.value
            , "ideas": []
        });
        console.dir(data);
        localStorage.setItem(app.key, JSON.stringify(data));
        getProfiles();
        inputName.value = '';
        inputDate.value = '';
        let close = document.getElementById("personModal");
        close.classList.remove("active");
    }
}

function personModalScreen() {
    console.log(app.currentContact);

    if (app.currentContact < 0) {
        personModal.querySelector(".title").innerHTML = "Person Add";
        inputDate.value = "";
        inputName.value = "";
    }
    else {
        personModal.querySelector(".title").innerHTML = "Person Edit";
        inputDate.value = data.people[app.currentContact].dob;
        inputName.value = data.people[app.currentContact].name;
    }
    let buttons = personModal.getElementsByTagName("button");
    let saveBtn = buttons[1];
    let cancelBtn = buttons[0];

    function cancelCheck() {
        console.log("cancel button");
        modal.classList.toggle("active");
        app.currentContact = -1;
        inputDate.value = "";
        inputName.value = "";
        cancelBtn.removeEventListener("touchstart", cancelCheck);
    }

    function saveCheck() {
        console.log("save button");
        if (app.currentContact > 0) {
            data.people[app.currentContact].name = inputName.value;
            data.people[app.currentContact].dob = inputDate.value;
            saveBtn.removeEventListener("touchstart", saveCheck);
        }
        else {
            data.people.push({
                "name": inputName.value
                , "dob": inputDate.value
                , "id": Date.now()
                , "ideas": []
            });
        }
        console.log(modal.className);

        inputName.value = "";
        app.currentContact = -1;
        console.dir(data.people);

        saveBtn.removeEventListener("touchstart", saveCheck);
    }
    cancelBtn.addEventListener("touchstart", cancelCheck());
